#define FINALIZADO                    0
#define NUMERO_DE_ARGUMENTOS          10
#define NUMERO_DE_ARGUMENTOS_INVALIDO 2
#define ARGUMENTO_INVALIDO            1
#define END_OF_STRING                 '\0'


int
protect (char arg[])
{
   int count;
   for(count = 0; arg[count] != END_OF_STRING;)
   {
      if(count != 0 )
      {
         return ARGUMENTO_INVALIDO;
      }
      count++;
   }

   return FINALIZADO;
}
